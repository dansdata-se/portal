# Switch

_Note: There exists a <LinkTo kind="components-switch--switch">helper component for switches</LinkTo>_

Switches are based on the [guidelines from Material Design 3](https://m3.material.io/components/switch/specs).

<Canvas>
  <Story
    name="Switch"
    args={{
      selection: false,
      thumb: "left",
      disabled: false,
      icon: false,
    }}
    argTypes={{
      thumb: {
        options: ["left", "right", "center"],
        control: {
          type: "radio",
        },
      },
    }}
  >
    {({ selection, thumb, disabled, icon }) => {
      const [state, setState] = useState(0);
      return (
        <button
          className="switch"
          data-selection={selection ? "true" : "false"}
          data-thumb={thumb}
          disabled={disabled}
          onClick={() => setState((state + 1) % 3)}
        >
          <div className="switch-track"></div>
          <div
            className={classNames("switch-thumb", {
              "switch-thumb-icon": icon,
            })}
          >
            {icon && <Check />}
          </div>
          <span className="sr-only">Switch state: {thumb}</span>
        </button>
      );
    }}
  </Story>
</Canvas>
